CREATE OR REPLACE PROCEDURE SP_CREATE_MESAS(v_id_mesa in NUMBER
,v_capacidad in NUMBER, v_estado_mesa NUMBER)
AS
BEGIN
  INSERT INTO MESAS VALUES(v_id_mesa,v_capacidad,v_estado_mesa);
END;


CREATE OR REPLACE PROCEDURE SP_R_ESTADO_MESA(estado out SYS_REFCURSOR)
IS 
BEGIN
    open estado for select*from estado_mesa;
END;

CREATE OR REPLACE PROCEDURE SP_READ_MESAS(prc out sys_refcursor)
AS
BEGIN
    OPEN prc for SELECT * FROM MESAS;
END;

CREATE OR REPLACE PROCEDURE SP_UPDATE_MESAS(id_mes in NUMBER,estado in NUMBER)
AS
BEGIN
    UPDATE MESAS SET ESTADO_MESA_ID_EST_MESA=estado WHERE id_mesa = id_mes;
END;
CREATE OR REPLACE PROCEDURE SP_DELETE_MESAS(id_mes in NUMBER)
AS
BEGIN
    DELETE FROM MESAS WHERE id_mesa = id_mes;
END;
